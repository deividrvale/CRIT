int f1(int n) {
  int r = 1;
  n = n/10;
  while(n>0){
    r++;
    n = n / 10;
  }
  return r;
}

int f2(int n) {
  int r = 1;
  n = n / 10;
  while( n > 0 ){
    r++;
    n = n / 10;
    if( n > 0 ){
      r++;
      n = n / 10;
      if( n > 0 ) {
	r++;
	n = n / 10;
	if( n > 0 ){
	  r++;
	  n = n / 10;
	}
      }
    }
  }
  return r;
}

      v = r;

      v = r+1;
      b = 0;
    }else if( n < 1000 ){
      v = r+2;
      b = 0;
    }else if( n < 10000 ){
      v = r+3;
      b = 0;
    }else{
      n = n / 10000;

    }
